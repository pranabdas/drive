<html>
    <head>
        <script>
        var jekyllPages = [];
        {% for post in site.posts %}jekyllPages.push("{{ site.url }}{{ site.baseurl }}{{ post.url }}");
        {% endfor %}
        {% for page in site.pages %}jekyllPages.push("{{ site.url }}{{ site.baseurl }}{{ page.url }}");
        {% endfor %}
        {% for item in site.static_files %}jekyllPages.push("{{ site.url }}{{ site.baseurl }}{{ item.path }}");
        {% endfor %}

        for (let jj = 0; jj < jekyllPages.length; jj++)  {
            if (jekyllPages[jj].slice(-1) === "/") {
                jekyllPages[jj] = jekyllPages[jj].slice(0, -1);
            }
            if (jekyllPages[jj].slice(-11).toLowerCase() === "/index.html"){
                jekyllPages[jj] = jekyllPages[jj].slice(0, -11);
            }
        }

        function jekyllFixCaseSensitiveUrl() {
            let url = window.location.toString();
            if (url.slice(-1) === "/"){
                url = url.slice(0, -1);
            }
            if (url.slice(-11).toLowerCase() === "/index.html") {
                url = url.slice(0, -11);
            }
            let pageList = jekyllPages.map(function(value) {
                return value.toLowerCase();
            });
            let i = pageList.indexOf(url.toLowerCase());
            if (i !== -1) {
                window.location = jekyllPages[i];
            } else {
                window.location.replace("{{ site.url }}/dev/null");
            }
        }
        if (window.location.toString().toLowerCase() !== "{{ site.url }}{{ site.baseurl}}/404.html") {
            window.onload = jekyllFixCaseSensitiveUrl();
        } else {
            window.location.replace("{{ site.url }}/404");
        }
        </script>
    </head>
</html>
